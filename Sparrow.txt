import javax.swing.*;
import java.awt.*;
import java.util.Random;

public class Sparrow extends Bird {
    public Sparrow() {
        System.out.println("воробей ворвался на плейс");
        color = Color.gray;
        size = 10;
    }
    public void draw(Graphics graphics){
        int min = z;
        int max = m;
        int diff = max - min;
        Random random = new Random();
        x = random.nextInt(diff + 1);
        x = x + min;
        int min1 = z;
        y = random.nextInt(diff + 1);
        y = y + min1;
        graphics.setColor(color);
        graphics.fillOval(x,y,size,size);
        if(x > Flock.xMax2){
            Flock.xMax2 = x;
        }
        if(x < Flock.xMin2){
            Flock.xMin2 = x;
        }
        if(y > Flock.yMax2){
            Flock.yMax2 = y;
        }
        if(y < Flock.yMin2){
            Flock.yMin2 = y;
        }
    }
}
